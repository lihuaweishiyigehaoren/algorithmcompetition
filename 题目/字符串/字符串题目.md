#### 最长回文子串

**描述**:

> <https://www.geeksforgeeks.org/longest-palindrome-substring-set-1/>
>
> 这里以动态规划方法解决这个问题，首先动态规划1、无后向性 2、最优子结构
>
> 最长回文子串问题：属于区间类动态规划问题，有通用的解题模式
>
> 通过存储子问题的结果可以减少复杂度

**源码：**

```c++
#include <iostream>
#include <string>
#include <vector>
using namespace std;

// 返回最长回文串的长度
int longestPalSubstr(string str)
{
	// 获取输入字符串的长度
	int n = str.size();

	// table[i][j]是false，如果i-j区间不是回文
	// 否则是true
	//bool table[n][n];
	vector<vector<bool>> table;
	for (int i = 0; i < n; i++)
	{
		table.emplace_back(vector<bool>(n, 0));
	}

	// 所有子串的长度1是回文
	int start = 0;
	int maxLength = 1;
	
	// 初始状态
	for (int i = 0; i < n - 1; i++)
	{
		if (str[i] == str[i + 1])
		{
			table[i][i + 1] = 1;
			start = i;
			maxLength = 2;
		}
	}

	// 检查长度超过2的，k是子串
	for (int k = 3; k <= n; ++k) //k是子串的长度，从3开始
	{
		// 确定开始的索引,从哪个位置开始
		for (int i = 0; i < n - k + 1; i++)
		{
			// 结束位置
			int j = i + k - 1;

			// 递推关系式
			if (table[i + 1][j - 1] && str[i] == str[j])
			{
				table[i][j] = true;
				if (k > maxLength)
				{
					start = i;
					maxLength = k;
				}
			}

		}
	}


	return maxLength;
}

int main() {
	string str = "forgeeksskeegfor";
	cout << "\nLength is: " << longestPalSubstr(str);
	system("pause");
	return 0;
}
```





#### 字符串相乘

**描述:**

>给定两个以字符串形式表示的非负整数 num1 和 num2，返回 num1 和 num2 的乘积，它们的乘积也表示为字符串形式。
>
>示例 1:
>
>输入: num1 = "2", num2 = "3"
>输出: "6"
>示例 2:
>
>输入: num1 = "123", num2 = "456"
>输出: "56088"
>说明：
>
>num1 和 num2 的长度小于110。
>num1 和 num2 只包含数字 0-9。
>num1 和 num2 均不以零开头，除非是数字 0 本身。
>不能使用任何标准库的大数类型（比如 BigInteger）或直接将输入转换为整数来处理。

**分析：**

><https://www.geeksforgeeks.org/multiply-large-numbers-represented-as-strings/>

**源码：**

```c++
#include<iostream>
#include<string> 
#include<vector>
using namespace std;

string calTwoString(string num1,string num2)
{
    // 计算ｎｕｍ1*num2(num2是被乘数)
    int len1 = num1.size();
    int len2 = num2.size();

    if(len1 ==0 || len2 == 0) return "0";

    // 存储结果的数组
    vector<int> result(len1+len2,0);// 结果是从低位到高位开始存储的
    
    int start_pos = 0; // start_pos是整体的位移，比如乘完被乘数的第一个数，该位置要+1，应为整体开始偏移位置+1了

    for(int i = len2-1;i>=0;i--)// num2
    {
        int curr_nums_i = num2[i]-'0';
        int num2_index = 0;
        int carry = 0;
        
        for(int j = len1-1;j>=0;j--)// num1
        {
            int curr_nums_j = num1[j]-'0';
            // 计算的是当前start_pos+num2_index位置的值
            int sum = curr_nums_i*curr_nums_j + result[start_pos+num2_index]+carry;
            result[start_pos+num2_index] = sum%10;
            carry = sum/10;
            num2_index++;
        }

        if(carry>0)
        {
            result[start_pos+num2_index]=carry;
        }

        start_pos++;
    }

    // 结果计算完毕
    int i = result.size()-1;
    while(i>=0&&result[i]==0) i--;
    if(i==-1) return "0";

    string s = "";
    while(i>=0)
    {
        s = s+std::to_string(result[i]);
        i--;
    }

    return s;
}

int main()
{
    string string1 = "1235421415454545454545454544";
    string string2 = "1714546546546545454544548544544545";

    cout<<calTwoString(string1,string2)<<endl;
    
    return 0;
}

```

#### 替换空格

**介绍解析：**

> 剑指offer P52



**源码：**

```c++
#include <iostream>
using namespace std;

// 蛮力遍历的方法不可取，因为复杂度高 n 个空格，就要移动后面的元素n次
// 该方法使用两个指针，从后往前面操作，提前把要移动的空格预留出来，那么就不会相同元素重复移动那么多次
void replaceSpace(char str[],int len)
{
	// 边界条件
	if (str == nullptr || len <= 0) return;

	// 遍历一遍字符串，确定要移动的总量是多少
	int spaceCount = 0;
	int origenLength = 0;
	/*while (*str != '\0')
	{
		origenLength++;
		if (*str == ' ') spaceCount++;
		str++;
	}*/
	int i = 0;
	while (str[i] != '\0')
	{
		origenLength++;
		if (str[i] == ' ') spaceCount++;
		i++;
	}

	int newLength = origenLength + spaceCount * 2;

	if (newLength > len) return;

	// 采用两个index，进行移位操作
	int p1 = origenLength;
	int p2 = newLength;

	while (p1>=0 && p2>p1)
	{
		if (str[p1] == ' ')
		{
			str[p1--] = '0';
			str[p1--] = '2';
			str[p1--] = '%';
		}
		else
		{
			// 如果不是空格，就把他放到p2指向的位置，同时移动p2的位置
			str[p2--] = str[p1];
		}
		p1--;
	}

}



int main()
{
	// 测试用例
	char str[100] = "wo shi yige haoren.";

	replaceSpace(str, 100);

	cout << str << endl;

	system("pause");

	return 0;
}
```



#### 找到字符串的最长无重复字符串

**说明：**

```c++
左程云P300
```

**源码：**

```c++
#include <iostream>
#include <string>
#include <math.h>
//#include <unordered_map>
using namespace std;
int max(int a, int b)
{
	return a > b ? a : b;
}

// 找到字符串的最长无重复字符子串
// 如果str长度为N,字符编码范围是M，本题可做到时间复杂度为O（N),额外空间复杂度为O（M）
int maxLengthOfStr(string str)
{
	if (str == "") return 0;

	//unordered_map<char, int> map;
	// 把字符当成ascii码
	int map[256];

	for (int i = 0; i < 256; i++)
	{
		map[i] = -1;
	}

	int length = str.length();
	const char * s = str.c_str();

	int len = 0;
	int pre = -1;
	int cur = 0;

	for (int i = 0; i < length; i++)
	{
		// 这里要取当前字符出现的最近的一个位置值和pre目前的值得最大值
		// pre变量表示，如果当前遍历到字符str[i]，pre表示必须以str[i-1]字符结尾的情况下，最长无重复字符子串开始位置的前一个位置
		// pre理解，加入现在是 a b c d，i的位置在d，pre就表示a的位置，就是最长无重复的地方
		// 假如现在是 a b c d b那么 pre的位置必须要更新到现在的b，然后记录这个最长

		// 确定pre的值
		pre = max(pre, map[s[i]]);
		// 当前长度cur
		cur = i - pre; // 第一个字母的时候长度是1
		len = max(len, cur);
		map[s[i]] = i;
	}

	return len;
}

int main()
{
	string testStr = "aabcb";
	int length = maxLengthOfStr(testStr);

	cout << length << endl;

	system("pause");
	return 0;
}
```
